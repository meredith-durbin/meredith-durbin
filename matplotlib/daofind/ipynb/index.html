from astropy.io import fits
import numpy as np
from matplotlib import pyplot as plt

# Get fits file info - this one is a drizzled image of NGC 6791
# made from 6 CTE-corrected UVIS exposures

image = fits.open('f606w_cts.fits')
data = image[0].data
image.close()

# Get coordinates from txdump'ed photometry file

f = open('f606w.phot')
coords = np.genfromtxt(f)
f.close()

xc = []
yc = []
mag = []
for i in coords:
    if i[3] > 0: #magfile has all INDEFs replaced by -999.0; remove these lines
        xc.append(i[0])
        yc.append(i[1])
        mag.append(i[3])

#Plot coordinates on top of image - basic version

fig = plt.figure()
ax = fig.add_subplot(111)
plt.set_cmap('bone')
ax.scatter(xc, yc, color='#00ff00')
ax.imshow(data, vmin=0, vmax=1000) #scale colormap so stars are easily visible
ax.set_xlim(1100,2500) #only plot a small section for illustration's sake
ax.set_ylim(6900,5500)
fig.tight_layout()

mag_scaled = np.mean(mag)/mag #scale to mean, account for magnitude scale being backward
